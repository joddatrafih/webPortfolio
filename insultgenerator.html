                    import random

                    # List setup for my adjectives and nouns
                    ADJECTIVES = ["contemptuous", "malevolent", "filthy", "disdainful", "petulant", "foul", "belligerent", "dimwitted", "repulsive", "weak-willed"]
                    NOUNS = ["donkey", "airhead", "dingeleberry", "doofus", "schmuck", "stumblebum"]

                    # Getting victims name
                    def getVictim():
                        while True:
                            name = input("Enter the victim's name: ")
                            if name != "":
                                return name
                            else:
                                print("Error! There must be a name!")

                    # Getting # of insults w/error checking for a non-int input
                    def getInsults():
                        while True:
                            try:
                                number = int(input("Enter the # of insults to generate: "))
                                if number > 0:
                                    return number
                                else:
                                    print("Error! # of insults must be 1 or greater!")
                            except ValueError:
                                print("Error! Please enter a valid integer!")

                    # Getting # adjectives w/error checking for a non-int input
                    def getAdjectives():
                        while True:
                            try:
                                number = int(input("Enter the # of adjectives in each insult (1, 2, or 3): "))
                                if number in [1, 2, 3]:
                                    return number
                                else:
                                    print("Error! Please enter 1, 2, or 3.")
                            except ValueError:
                                print("Error! Please enter a valid integer!")

                    # Generating the insult string to output to main
                    def genInsults(victim, insultNumber, adjectiveNumber):
                        for n in range(insultNumber):

                            # Building a list of random adjectives 
                            adjectives = []
                            for n in range(adjectiveNumber):
                                adjectives += [f"\033[34m{random.choice(ADJECTIVES)}\033[0m"]
                            
                            # Selecting a random noun
                            noun = random.choice(NOUNS)
                            
                            # Manually constructing the insult w/.join for spacing
                            insult = (f"{victim} is a " + " ".join(adjectives) + " " + noun + "!")
                            
                            print(insult)

                    # Loop to replay or not
                    def replay():
                        while True:
                            choice = input("Do you want to play again? (y/n): ")
                            if choice.lower() in ['y', 'n']:
                                return choice.lower() == 'y'
                            print("Error: Please enter 'y' or 'n'")

                    # Main function to call the above and generate the insults
                    def main():
                        while True:
                            
                            # Generate the insult
                            genInsults(getVictim(), getInsults(), getAdjectives())
                            
                            # Ask if the user wants to play again
                            if not replay():
                                print("You took it like a champ!")
                                break

                    main()
